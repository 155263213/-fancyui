{{'Example'|translate}}

view

controller

export default class SelectController { constructor() { this.city = {}; this.data = [ { id: '1001', name: '上海' }, { id: '1002', name: '北京' }, { id: '1003', name: '广州' }, { id: '1004', name: '杭州' }, { id: '1005', name: '深圳' } ] } onSelect($item) { console.log($item) } }

有禁用选项

{{'Example'|translate}}

view

controller

export default class SelectController { constructor() { this.city = {}; this.data = [ { id: '1001', name: '上海' }, { id: '1002', name: '北京' }, { id: '1003', name: '广州' }, { id: '1004', name: '杭州' }, { id: '1005', name: '深圳', disabled: true } ] } onSelect($item) { console.log($item) } }

禁用

选择器不可用状态

{{'Example'|translate}}

view

<f-select data="vm.data" ng-model="vm.city" is-disabled = "true"> </f-select>

controller

export default class SelectController { constructor() { this.data = [ { id: '1001', name: '上海' }, { id: '1002', name: '北京' }, { id: '1003', name: '广州' }, { id: '1004', name: '杭州' }, { id: '1005', name: '深圳', disabled: true } ] this.city = this.data[0]; } }

可清空选项

包含清空按钮，可将选择器清空为初始状态

{{'Example'|translate}}

view

<f-select data="vm.data" ng-model="vm.city" clearable="true" on-item-selected="vm.onSelect($item)" </f-select>

controller

export default class SelectController { constructor() { this.data = [ { id: '1001', name: '上海' }, { id: '1002', name: '北京' }, { id: '1003', name: '广州' }, { id: '1004', name: '杭州' }, { id: '1005', name: '深圳', disabled: true } ] this.city = {}; } }

基础多选

适用性较广的基础多选，用 Tag 展示已选项

{{'Example'|translate}}

view

<f-select data="vm.data" ng-model="vm.city" is-multiple="true" on-item-selected="vm.onSelect($item)" </f-select>

controller

export default class SelectController { constructor() { this.data = [ { id: '1001', name: '上海江桥万达广场' }, { id: '1002', name: '北京通州万达广场' }, { id: '1003', name: '上海周浦万达广场' }, { id: '1004', name: '武汉万达广场' }, { id: '1005', name: '浙江椒江万达广场' } ]; this.city = {}; } }
