安装fancyui

注入模块

import radioComponent from './radio.component'; import {Radiobox} from 'fancyui'; import {RadioboxGroup} from 'fancyui'; export default angular.module('radio', [ Radiobox.name, RadioboxGroup.name ]) .component('radio', radioComponent);
