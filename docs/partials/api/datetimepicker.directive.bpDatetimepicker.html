bpDatetimepicker
directive in module datetimepicker

Description

  • 时间选择组件
  • 支持单个时间框和一组时间框

Usage

as element:
<bp-datetimepicker
       type="{String}"
       model="{String}"
       start-model="{String}"
       end-model="{String}"
       min="{String | Number}"
       max="{String | Number}"
       all-disabled="{Boolean}"
       start-disabled="{Boolean}"
       end-disabled="{Boolean}"
       placeholders="{Array}"
       model-type="{String}"
       minute-step="{Number}"
       min-view="{String}"
       start-view="{String}">
</bp-datetimepicker>

Parameters

ParamTypeDetails
typeString
  • binding symbol is <, 判断展示单个时间框或一组时间框
modelString
  • binding symbol is =?, 存储单个时间框的值
startModelString
  • binding symbol is =?, 存储一组时间框的开始时间
endModelString
  • binding symbol is =?, 存储一组时间框的结束时间
minString Number
  • binding symbol is <, 最小值
maxString Number
  • binding symbol is <, 最大值
allDisabledBoolean
  • binding symbol is <, 是否禁用全部 默认为 false 不禁用
startDisabledBoolean
  • binding symbol is <, 是否禁用开始时间 默认为 false 不禁用
endDisabledBoolean
  • binding symbol is <, 是否禁用结束时间 默认为 false 不禁用
placeholdersArray
  • binding symbol is <, 文本框从左到右的placeholder 默认为 ['请选择开始时间', '请选择结束时间']
modelTypeString
  • binding symbol is <, 时间展示格式 默认为 'YYYY-MM-DD HH:mm:ss'
minuteStepNumber
  • binding symbol is <, 分钟间隔 默认为 1
minViewString
  • binding symbol is <, 最小视图 默认为 'minute'
startViewString
  • binding symbol is <, 首次打开时的视图 默认为 'day'

Example

1、默认的单个时间框 

  <bp-datetimepicker model="vm.config.model"></bp-datetimepicker>

  this.config = {
    model: ""
  };
2、禁用单个时间框 
<bp-datetimepicker
  model="vm.newConfig.model"
  placeholders="vm.newConfig.placeholders"
  all-disabled="vm.newConfig.allDisabled"
></bp-datetimepicker>

this.newConfig = {
  model: "",
  placeholders: ['单个时间框'],
  allDisabled: true
};
3、默认的一组时间框（纵向） 
<div class="row">
  <div class="col-xs-6">
    <div class="form-group">
      <label>纵向时间框排列</label>
      <bp-datetimepicker
        type="vm.config.type"
        start-model="vm.config.startModel"
        end-model="vm.config.endModel"
      ></bp-datetimepicker>
    </div>
  </div>
</div>

this.config = {
  type: "range",
  startModel: "",
  endModel: ""
};
4、默认的一组时间框（横向） 
<form class="form-horizontal" novalidate>
  <div class="form-group">
    <label class="col-sm-2 control-label">横向时间框排列</label>
    <div class="col-sm-5">
      <bp-datetimepicker
        type="vm.config.type"
        start-model="vm.config.startModel"
        end-model="vm.config.endModel"
      ></bp-datetimepicker>
    </div>
  </div>
</form>

this.config = {
  type: "range",
  startModel: "",
  endModel: ""
};
5、限制时间范围的时间框 
<form class="form-horizontal" novalidate>
  <div class="form-group">
    <div class="col-sm-8">
      <bp-datetimepicker
        type="vm.config.type"
        start-model="vm.config.startModel"
        end-model="vm.config.endModel"
        min="vm.config.min"
        max="vm.config.max"
        placeholders="vm.config.placeholders"
      ></bp-datetimepicker>
    </div>
  </div>
</form>

this.config = {
  type: "range",
  startModel: "",
  endModel: "",
  min: "2016-7-23 14:29:39", // Date()、时间戳、moment()、"today"
  max: "today",              // Date()、时间戳、moment()、"today"
  placeholders: ['启动时间', '结束时间']
};
6、全部禁用时间框 
<form class="form-horizontal" novalidate>
  <div class="form-group">
    <div class="col-sm-8">
      <bp-datetimepicker
        type="vm.newConfig.type"
        start-model="vm.newConfig.startModel"
        end-model="vm.newConfig.endModel"
        all-disabled="vm.newConfig.allDisabled"
      ></bp-datetimepicker>
    </div>
  </div>
</form>

this.newConfig = {
  type: "range",
  startModel: moment().format('YYYY-MM-DD HH:mm:ss'),
  endModel: "",
  allDisabled: true
};
7、禁用开始时间框 
<form class="form-horizontal" novalidate>
  <div class="form-group">
    <div class="col-sm-8">
      <bp-datetimepicker
        type="vm.startDisabledConfig.type"
        start-model="vm.startDisabledConfig.startModel"
        end-model="vm.startDisabledConfig.endModel"
        start-disabled="vm.startDisabledConfig.startDisabled"
        placeholders="vm.startDisabledConfig.placeholders"
        start-view="vm.startDisabledConfig.startView"
        min-view="vm.startDisabledConfig.minView"
        model-type="vm.startDisabledConfig.modelType"
      ></bp-datetimepicker>
    </div>
  </div>
</form>

this.startDisabledConfig = {
  type: "range",
  startModel: moment().format('YYYY-MM-DD'),
  endModel: "",
  placeholders: ['', '禁用开始时间框'],
  startDisabled: true,
  startView: "month",
  minView: "day",
  modelType: "YYYY-MM-DD"
};
8、禁用结束时间框 
<form class="form-horizontal" novalidate>
  <div class="form-group">
    <div class="col-sm-8">
      <bp-datetimepicker
        type="vm.endDisabledConfig.type"
        start-model="vm.endDisabledConfig.startModel"
        end-model="vm.endDisabledConfig.endModel"
        end-disabled="vm.endDisabledConfig.endDisabled"
        placeholders="vm.endDisabledConfig.placeholders"
        start-view="vm.endDisabledConfig.startView"
        min-view="vm.endDisabledConfig.minView"
      ></bp-datetimepicker>
    </div>
  </div>
</form>

this.endDisabledConfig = {
  type: "range",
  startModel: "",
  endModel: moment().format('YYYY-MM-DD HH:mm:ss'),
  placeholders: ['禁用结束时间框'],
  endDisabled: true,
  startView: "month",
  minView: "hour",
};
